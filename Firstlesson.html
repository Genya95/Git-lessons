package ru.geekbrains.lesson1; public class FirstApp { public static void main(String[] args) { System.out.println("Orange"); System.out.println("Banana"); System.out.println("Apple"); checkSumSign(); printColour(); compareNumbers(); } public static void checkSumSign() { int a = 15; int b = 31; int c = a + b; System.out.println("a + b =" + c); if (c >= 0) System.out.println("Сумма положительна"); if (c < 0) System.out.println("Сумма отрицательна"); } public static void printColour() { int value = -13; System.out.println("value =" + value); if (value <= 0) System.out.println("Красный"); if (value > 0 && value <= 100) System.out.println("Желтый"); if (value > 100) System.out.println("Зеленый"); } public static void compareNumbers() { int a = 60; int b = -140; if (a >= b) { System.out.println("a >= b"); } else { System.out.println("a < b"); } } }
